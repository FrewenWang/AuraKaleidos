if (${PRODUCT} MATCHES "mainline")
    collect_source_files(lib_srcs GLOB manager)
elseif (${PRODUCT} MATCHES "jidu_mobile")
    list(APPEND lib_srcs ${SRC_ROOT}/manager/AbsVisionManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/ImageBrightnessManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/FaceRectManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/FaceLandmarkManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/FaceQualityManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/FaceInteractLivingManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/FaceFeatureManager.cpp)
    list(APPEND lib_srcs ${SRC_ROOT}/manager/VisionManagerRegistry.cpp)
else ()
    collect_source_files(manager_srcs GLOB manager)
    foreach (src ${manager_srcs})
        get_filename_component(src_name ${src} NAME)
        set(customize_src ${SRC_ROOT}/manager/customize/${PRODUCT}/${src_name})
        if (EXISTS ${customize_src})
            list(APPEND lib_srcs ${customize_src})
            message(STATUS "[${PRODUCT}]: using customized manager: ${customize_src}")
        else ()
            list(APPEND lib_srcs ${src})
        endif ()
    endforeach ()
endif ()

if (BUILD_EXPERIMENTAL)
    collect_source_files(lib_srcs GLOB manager/experimental)
endif ()
