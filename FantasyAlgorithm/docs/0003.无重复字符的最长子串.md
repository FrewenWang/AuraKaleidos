class Solution {
public:
  int lengthOfLongestSubstring(string s) {
    int n = s.length();
    int left = 0, right = 0;
    int ans = 0;
    // c++中的
    unordered_set<char> set;
    for (; right < n; ++right) {
      // 判断右侧zhi
      char ch = s[right];
      while (set.count(ch)) {
        char c = s[left++];
        set.erase(c);
      }
      set.insert(ch);
      ans = std::max(ans, right - left + 1);
    }
    return ans;
  }
};
