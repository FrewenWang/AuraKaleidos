if (${PRODUCT} MATCHES "mainline")
    collect_source_files(lib_srcs GLOB scheduler)
    collect_source_files(lib_srcs GLOB scheduler/dag)
else ()
    collect_source_files(scheduler_srcs GLOB scheduler)
    collect_source_files(scheduler_srcs GLOB scheduler/dag)
    foreach (src ${scheduler_srcs})
        get_filename_component(src_name ${src} NAME)
        set(customize_src ${SRC_ROOT}/scheduler/customize/${PRODUCT}/${src_name})
        if (EXISTS ${customize_src})
            list(APPEND lib_srcs ${customize_src})
            message(STATUS "[${PRODUCT}]: using customized scheduler: ${customize_src}")
        else ()
            list(APPEND lib_srcs ${src})
        endif ()
    endforeach ()
endif ()
