message(STATUS "[vision:config] product=" ${PRODUCT})
if (${PRODUCT} MATCHES "mainline")
    collect_source_files(lib_srcs GLOB config/runtime_config)
else ()
    collect_source_files(manager_srcs GLOB config/runtime_config)
    foreach (src ${manager_srcs})
        get_filename_component(src_name ${src} NAME)
        set(customize_src ${SRC_ROOT}/config/runtime_config/customize/${PRODUCT}/${src_name})
        if (EXISTS ${customize_src})
            list(APPEND lib_srcs ${customize_src})
            message(STATUS "[${PRODUCT}]: using customized manager: ${customize_src}")
        else ()
            list(APPEND lib_srcs ${src})
        endif ()
    endforeach ()
endif ()
